class Wizard:
    def __init__(self, name, surname, level, type_of_magic_wand, side_of_the_world_of_magic, magic_pet,
                 magical_abilities, blood_purity, role="—Å—Ç—É–¥–µ–Ω—Ç", appearance=None):
        self.name = name
        self.surname = surname
        self.level = level
        self.type_of_magic_wand = type_of_magic_wand
        self.side_of_the_world_of_magic = side_of_the_world_of_magic
        self.magic_pet = magic_pet
        self.magical_abilities = magical_abilities
        self.blood_purity = blood_purity
        self.role = role
        self.health = 100
        self.reputation_points = 100
        self.max_health = 100
        self.wins = 0
        self.losses = 0
        self.artifacts = []
        self.max_artifacts = 1
        self.hogwarts_points = 0
        self.lessons_attended = 0
        self.appearance = appearance if appearance else random.choice(variations)
        self.galleons = random.randint(50, 200)

    def take_damage(self, amount):
        self.health -= amount
        if self.health < 0:
            self.health = 0
        print(f"{self.name} –ø–æ–ª—É—á–∏–ª {amount} —É—Ä–æ–Ω–∞. –û—Å—Ç–∞–ª–æ—Å—å –∑–¥–æ—Ä–æ–≤—å—è: {self.health}")

    def restore_health(self, amount):
        self.health += amount
        if self.health > self.max_health:
            self.health = self.max_health
        print(f"{self.name} –≤–æ—Å—Å—Ç–∞–Ω–æ–≤–∏–ª {amount} –∑–¥–æ—Ä–æ–≤—å—è. –¢–µ–∫—É—â–µ–µ –∑–¥–æ—Ä–æ–≤—å–µ: {self.health}")

    def lose_reputation(self, points):
        self.reputation_points -= points
        if self.reputation_points < 0:
            self.reputation_points = 0
        print(f"{self.name} –ø–æ—Ç–µ—Ä—è–ª {points} –ø—É–Ω–∫—Ç–æ–≤ —Ä–µ–ø—É—Ç–∞—Ü–∏–∏. –û—Å—Ç–∞–ª–æ—Å—å —Ä–µ–ø—É—Ç–∞—Ü–∏–∏: {self.reputation_points}")

    def gain_reputation(self, points):
        self.reputation_points += points
        print(f"{self.name} –∑–∞—Ä–∞–±–æ—Ç–∞–ª {points} –ø—É–Ω–∫—Ç–∞ —Ä–µ–ø—É—Ç–∞—Ü–∏–∏. –í—Å–µ–≥–æ —Ä–µ–ø—É—Ç–∞—Ü–∏–∏: {self.reputation_points}")

    def get_available_artifacts_count(self):
        if self.reputation_points >= 300:
            return 5
        elif self.reputation_points >= 250:
            return 4
        elif self.reputation_points >= 200:
            return 3
        elif self.reputation_points >= 150:
            return 2
        else:
            return 1

    def can_equip_artifact(self):
        return len(self.artifacts) < self.get_available_artifacts_count()

    def add_artifact(self, artifact):
        if self.can_equip_artifact():
            self.artifacts.append(artifact)
            print(f"{self.name} –ø–æ–ª—É—á–∏–ª –∞—Ä—Ç–µ—Ñ–∞–∫—Ç: {artifact.name}!")
            return True
        else:
            print(f"{self.name} –Ω–µ –º–æ–∂–µ—Ç –Ω–æ—Å–∏—Ç—å –±–æ–ª—å—à–µ –∞—Ä—Ç–µ—Ñ–∞–∫—Ç–æ–≤! –¢—Ä–µ–±—É–µ—Ç—Å—è –ø–æ–≤—ã—à–µ–Ω–∏–µ —Ä–µ–ø—É—Ç–∞—Ü–∏–∏.")
            return False

    def use_artifact_in_battle(self):
        if not self.artifacts:
            return 0

        strongest_artifact = max(self.artifacts, key=lambda x: x.power)
        print(f"{self.name} –∏—Å–ø–æ–ª—å–∑—É–µ—Ç –∞—Ä—Ç–µ—Ñ–∞–∫—Ç {strongest_artifact.name}!")
        return strongest_artifact.power

    def display_appearance(self):
        if self.appearance:
            print(f"–í–Ω–µ—à–Ω–æ—Å—Ç—å {self.name}:")
            for line in self.appearance:
                print(line)
            print()

    def wizard_info(self):

        print(f"–°—Ç–æ—Ä–æ–Ω–∞ –≤ –º–∏—Ä–µ –º–∞–≥–∏–∏: {self.side_of_the_world_of_magic}")
        print(f"–ú–∞–≥–∏—á–µ—Å–∫–∏–π –ø–∏—Ç–æ–º–µ—Ü: {self.magic_pet}")
        print(f"–ú–∞–≥–∏—á–µ—Å–∫–∏–µ –Ω–∞–≤—ã–∫–∏: {self.magical_abilities}")
        print(f"–ß–∏—Å—Ç–æ—Ç–∞ –∫—Ä–æ–≤–∏: {self.blood_purity}")
        print(f"–ó–¥–æ—Ä–æ–≤—å–µ: {self.health}/{self.max_health}")
        print(f"–†–µ–ø—É—Ç–∞—Ü–∏—è: {self.reputation_points}")
        print(f"–ì–∞–ª–ª–µ–æ–Ω—ã: {self.galleons}")

        print(f"–ü–æ–±–µ–¥—ã/–ü–æ—Ä–∞–∂–µ–Ω–∏—è: {self.wins}/{self.losses}")
        print(f"–ê—Ä—Ç–µ—Ñ–∞–∫—Ç—ã: {len(self.artifacts)}/{self.get_available_artifacts_count()}")
        for artifact in self.artifacts:
            print(f"  - {artifact.name} (—Å–∏–ª–∞: {artifact.power})")
        self.display_appearance()
        print(" ~ * ~ " * 8)


def show_wizard_appearances():
    print("\n" + "=" * 100)
    print("–í–´–ë–ï–†–ò–¢–ï –í–ù–ï–®–ù–û–°–¢–¨ –í–ê–®–ï–ì–û –í–û–õ–®–ï–ë–ù–ò–ö–ê")
    print("=" * 100)

    for row in range(5):
        for i, wizard in enumerate(variations, 1):
            if i < 10:
                print(f"{i}. {wizard[row]}  ", end="")
            else:
                print(f"{i}.{wizard[row]}  ", end="")
        print()

class Battle:
    def __init__(self, player_wizard, opponent_wizard):
        self.player_wizard = player_wizard
        self.opponent_wizard = opponent_wizard

    def display_battle_scene(self):
        print("\n" + "=" * 60)
        print(f"–ë–ò–¢–í–ê: {self.player_wizard.name} vs {self.opponent_wizard.name}")
        print("=" * 60)

        battle_symbols = [
            "  ‚ö°  ",
            "  ‚ú®  ",
            "  üî•  ",
            "  üí•  ",
            "  üåü  "
        ]

        battle_symbol = random.choice(battle_symbols)

        for i in range(5):
            player_line = self.player_wizard.appearance[i] if i < len(self.player_wizard.appearance) else "     "
            opponent_line = self.opponent_wizard.appearance[i] if i < len(self.opponent_wizard.appearance) else "     "

            print(f"{player_line}  {battle_symbol}  {opponent_line}")

        player_health_bar = "‚ñà" * (self.player_wizard.health // 10) + "‚ñë" * (10 - self.player_wizard.health // 10)
        opponent_health_bar = "‚ñà" * (self.opponent_wizard.health // 10) + "‚ñë" * (10 - self.opponent_wizard.health // 10)

        print(f"\n{self.player_wizard.name}: {player_health_bar} {self.player_wizard.health} HP")
        print(f"{self.opponent_wizard.name}: {opponent_health_bar} {self.opponent_wizard.health} HP")
        print("=" * 60)

    def fight(self):
        self.display_battle_scene()

        print(f"\n–ë–æ–π –Ω–∞—á–∞–ª—Å—è:\n{self.player_wizard.name} VS {self.opponent_wizard.name}")

        round_num = 1
        while self.player_wizard.health > 0 and self.opponent_wizard.health > 0:
            print(f"\n--- –†–∞—É–Ω–¥ {round_num} ---")

            self.display_battle_scene()

            action = input(
                f"–•–û–î {self.player_wizard.name}: –ê—Ç–∞–∫–æ–≤–∞—Ç—å(–∞) / –ó–∞—â–∏—Ç–∏—Ç—å—Å—è(–∑) / –û—Å–æ–±–æ–µ —É–º–µ–Ω–∏–µ(–æ) / –ò—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å –∞—Ä—Ç–µ—Ñ–∞–∫—Ç(–∞—Ä)? ").lower()
            if action == '–∞':
                damage = random.randint(15, 35)
                self.opponent_wizard.take_damage(damage)
            elif action == '–∑':
                healing = random.randint(10, 20)
                self.player_wizard.restore_health(healing)
            elif action == '–æ':
                special_damage = random.randint(25, 45)
                print(f"{self.player_wizard.name} –∏—Å–ø–æ–ª—å–∑—É–µ—Ç {self.player_wizard.magical_abilities}!")
                self.opponent_wizard.take_damage(special_damage)
            elif action == '–∞—Ä':
                artifact_bonus = self.player_wizard.use_artifact_in_battle()
                if artifact_bonus > 0:
                    self.opponent_wizard.take_damage(artifact_bonus)
            else:
                print("–ù–µ–≤–µ—Ä–Ω–æ–µ –¥–µ–π—Å—Ç–≤–∏–µ! –ü—Ä–æ–ø—É—Å–∫–∞–µ—Ç–µ —Ö–æ–¥.")

            if self.opponent_wizard.health <= 0:
                break

            print(f"\n–•–û–î {self.opponent_wizard.name}:")
            comp_action = random.choice(['–∞', '–∑', '–æ', '–∞—Ä'])
            if comp_action == '–∞':
                damage = random.randint(15, 35)
                self.player_wizard.take_damage(damage)
            elif comp_action == '–∑':
                healing = random.randint(10, 20)
                self.opponent_wizard.restore_health(healing)
            elif comp_action == '–æ':
                special_damage = random.randint(25, 45)
                print(f"{self.opponent_wizard.name} –∏—Å–ø–æ–ª—å–∑—É–µ—Ç {self.opponent_wizard.magical_abilities}!")
                self.player_wizard.take_damage(special_damage)
            elif comp_action == '–∞—Ä':
                artifact_bonus = self.opponent_wizard.use_artifact_in_battle()
                if artifact_bonus > 0:
                    self.player_wizard.take_damage(artifact_bonus)

            round_num += 1

        if self.player_wizard.health > 0:
            print(f"\n{self.player_wizard.name} –ø–æ–±–µ–¥–∏–ª!")
            reputation_gain = random.randint(20, 40)
            self.player_wizard.gain_reputation(reputation_gain)
            self.opponent_wizard.lose_reputation(reputation_gain // 2)
            self.player_wizard.wins += 1
            self.opponent_wizard.losses += 1
            self.award_winner(self.player_wizard)
        else:
            print(f"\n{self.opponent_wizard.name} –ø–æ–±–µ–¥–∏–ª!")
            reputation_gain = random.randint(20, 40)
            self.opponent_wizard.gain_reputation(reputation_gain)
            self.player_wizard.lose_reputation(reputation_gain // 2)
            self.opponent_wizard.wins += 1
            self.player_wizard.losses += 1
            self.award_winner(self.opponent_wizard)

    def award_winner(self, winner):
        artifacts_pool = create_artifacts()
        available_artifacts = [a for a in artifacts_pool if a not in winner.artifacts]

        if available_artifacts and winner.can_equip_artifact():

        money_reward = random.randint(20, 50)
